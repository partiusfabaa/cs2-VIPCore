```json
        "weapon_hegrenade": 1,
        "weapon_incgrenade": 1~~~~~~~~
    },
    "T": {
        "weapon_flashbang": 2,
        "weapon_smokegrenade": 1,
        "weapon_hegrenade": 1,
        "weapon_molotov": 1
    }
}
```
